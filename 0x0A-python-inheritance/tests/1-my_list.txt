import unittest

class TestMyList(unittest.TestCase):
    def test_print_sorted(self):
        # Test if print_sorted prints the list in ascending order
        my_list = MyList([3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5])
        expected_output = [1, 1, 2, 3, 3, 4, 5, 5, 5, 6, 9]

        # Redirect stdout to capture the printed output
        import sys
        from io import StringIO
        original_stdout = sys.stdout
        sys.stdout = StringIO()

        try:
            my_list.print_sorted()
            printed_output = sys.stdout.getvalue().strip()
            self.assertEqual(printed_output, str(expected_output))
        finally:
            sys.stdout = original_stdout

if __name__ == '__main__':
    unittest.main()
